		function _sortByZIndex(a) {
			var k, i;
			for (i = 0; i < a.length - 1; i++) {
				k = i;
				for (j = i + 1; j < a.length; j++)
					if (a[k].css("z-index") > a[j].css("z-index"))
						k = j;
				if (k != i) {
					t = a[i];
					a[i] = a[k];
					a[k] = t;
				}
			}
		}

	function assignRoom(element, roomId, roomNo, houseorientation) {
		if (!isSelect) {
			return;
		}
		isSelect = false;
		var roomInfo = $(element).find(".room-info").html();
		$.ajax({
			type: "POST",
			url: "/ModuleBook/PersonSelectRoom/AjaxAssignRoom",
			data: "roomId=" + roomId,
			success: function (msg) {
				msg = eval("(" + msg + ")");
				if (msg.result == "") {
					removeRoomSelected();
					$(element).removeClass("room-available").addClass('room-selected ');
					$(element).html("");
					$(element).append("<span class='room-no'>" + roomNo + "朝向：" + houseorientation + "</span>");
					$(element).append("<div class='clear'></div>");
					$(element).append("<div class='room-info'>已预定</div>");
					$(element).fadeOut();
					$(element).fadeIn();
					isSelect = true;
					parent.SetValue(msg.roomid, msg.roomtpeyid, roomNo, msg.roomtypename, msg.conmunityID); //回参

				}
				else {
					isSelect = true;
					window.top.sysAlert(msg.result);
				}
			}
		});
	}


	function unAssign(element, roomId, roomNo, houseorientation) {

		$.ajax({
			type: "POST",
			url: "/ModuleRent/PersonalSelectRoom/AjaxUnAssignRoom",
			data: "ApplyId=" + applyId + "&roomId=" + roomId,
			success: function (msg) {
				msg = eval("(" + msg + ")");
				if (msg.result == "") {
					$(element).removeClass().addClass('room-available');
					$(element).html("");
					$(element).append("<span class='room-no'>" + roomNo + "朝向：" + houseorientation + "</span>");
					$(element).append("<div class='clear'></div>");
					$(element).append("<div class='room-info'>" + (msg.info == "" ? element.attr("infor") : msg.info) + "</div>");
				}
				else {
					window.top.sysAlert(msg.result);
				}
			}
		});
	}

	function removeRoomSelected() {
		var element = $(".room-selected");
		element.removeClass().addClass('room-available');
		element.html("");
		element.append("<span class='room-no'>" + element.attr("room-no") + "朝向:" + element.attr("houseorientation") + "</span>");
		element.append("<div class='clear'></div>");
		element.append("<div class='room-info'>" + element.attr("infor") + "</div>");
	}

	function operateRoom(element, roomId, roomNo, houseorientation) {
		if ($(element).children(":first").attr("class").indexOf("room-selected") >= 0) {
			unAssign($(element).children(":first"), roomId, roomNo, houseorientation);
			return;
		}

		if ($(element).children(":first").attr("class").indexOf("room-available") >= 0) {
			assignRoom($(element).children(":first"), roomId, roomNo, houseorientation);
			return;
		}
	}
